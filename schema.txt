TABLES

users
----------
userID                                      SERIAL
email                                       VARCHAR(255)
password (hashed)                           VARCHAR(255)
date                                        DATE
isAdmin                                     BOOL

-> CREATE TABLE users (userID SERIAL, email VARCHAR(255), 
username VARCHAR(30), password VARCHAR(255), date DATE, isAdmin BOOL);

ALTER SEQUENCE users_userid_seq RESTART WITH 1;

posts
----------
postID                                      SERIAL
authorID (join with userId)                 int
title                                       VARCHAR(50)
content                                     text
likes                                       int
published                                   DATE

-> CREATE TABLE posts (postID SERIAL, authorID int, title VARCHAR(50), 
content VARCHAR(1000), likes int, published DATE);

comments
-----------
commentID                                   SERIAL
authorID (joined with postId and userId)    int
postID (joined with postId and userId)      int
content                                     VARCHAR(300)
likes                                       int
published                                   DATE

-> CREATE TABLE comments (commentID SERIAL, authorID int, postID int, 
content VARCHAR(300), likes int, published DATE);
